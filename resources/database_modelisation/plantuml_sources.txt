    {field} superficy(km²) : BINARY_FLOAT
}

    {field} #id_dst : NUMBER

}

    {field} code : VARCHAR(3)
}

table(ICAO) {
    {field} #id_icao : NUMBER
    {field} code : VARCHAR(4)
}

table(PLANE_TYPE) {
    {field} #id_plane_type : NUMBER
    {field} type : VARCHAR(50)
    {field} passengers : NUMBER
    {field} freight(Kg) : NUMBER
    {field} consumption(L/100 km) : NUMBER

}

    {field} #id_timezone : NUMBER
    {field} from_cte : NUMBER

table(AIRLINE) {
    {field} #id_airline : NUMBER
    {field} +id_country : NUMBER
    {field} active : BOOL
    {field} alias : VARCHAR(50)
    {field} callsing : VARCHAR(50)
    {field} name : VARCHAR(100)
}

    {field} #id_city : NUMBER
    {field} +id_country : NUMBER

    {field} name :VARCHAR(100)
}

    {field} #id_fly_on : NUMBER

    {field} +id_plane : NUMBER

table(AIRPORT) {
    {field} #id_airport : NUMBER
    {field} +id_city : NUMBER
    {field} +id_iata : NUMBER
    {field} +id_icao : NUMBER
    {field} altitude : BINARY_FLOAT
    {field} latitude : BINARY_FLOAT
    {field} longitude : BINARY_FLOAT
    {field} name : VARCHAR(150)
}

table(PLANE) {
    {field} #id_plane : NUMBER
    {field} +id_airline : NUMBER
    {field} +id_plane_type : NUMBER
    {field} +id_iata : NUMBER
    {field} icao_code : VARCHAR(4)
}

table(AIRWAY) {
    {field} #id_airway : NUMBER
    {field} +id_airline : NUMBER

    {field} +id_airport_src : NUMBER
    {field} +id_equipment : NUMBER
    {field} +id_plane_type : NUMBER

}
